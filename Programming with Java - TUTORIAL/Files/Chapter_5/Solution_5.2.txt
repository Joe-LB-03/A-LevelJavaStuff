public class OddsAndEvens{
	public static void main(String[] args){

           	System.out.println("ODDS:");
		for(int i=1; i<101; i=i+2){
              		System.out.println(i);
           	}

           	System.out.println("EVENS:"); 
           	for(int i=2; i<101; i=i+2){
              		System.out.println(i);
           	}
      }
}
